\documentclass[border=10pt]{standalone}
\usepackage{tikz}
\usetikzlibrary{arrows.meta, positioning, fit, decorations.pathmorphing}

\begin{document}

\definecolor{ctrlcolor}{RGB}{0, 120, 200}
\definecolor{espcolor}{RGB}{120, 80, 180}
\definecolor{apicolor}{RGB}{50, 140, 50}
\definecolor{gray1}{RGB}{120, 120, 120}
\begin{tikzpicture}[
    box/.style={
      rectangle, rounded corners=4pt, draw=#1, line width=1.5pt,
      fill=#1!8, minimum width=40mm, minimum height=18mm,
      font=\large\bfseries, text=#1, align=center
    },
    smallbox/.style={
      rectangle, rounded corners=2pt, draw=#1, line width=1pt,
      fill=#1!6, minimum width=32mm, minimum height=12mm,
      font=\bfseries, text=#1, align=center
    },
    arr/.style={-{Stealth[length=3mm]}, line width=1.5pt, #1},
    lbl/.style={font=\small, text=gray1, align=center},
  ]

  % Controller side
  \node[box=ctrlcolor] (ctrl) at (0, 0) {Controller};

  % Sticks (2x2 grid with axis labels inside)
  \coordinate (stickcenter) at ([yshift=-38mm]ctrl.south);
  \node[smallbox=ctrlcolor, minimum height=14mm, left=6mm of stickcenter, yshift=12mm]
    (lstick) {Throttle\\[-1pt]{\scriptsize\color{gray1} Left Stick Y}};
  \node[smallbox=ctrlcolor, minimum height=14mm, right=6mm of stickcenter, yshift=12mm]
    (rstick_r) {Roll\\[-1pt]{\scriptsize\color{gray1} Right Stick X}};
  \node[smallbox=ctrlcolor, minimum height=14mm, left=6mm of stickcenter, yshift=-12mm]
    (lstick_y) {Yaw\\[-1pt]{\scriptsize\color{gray1} Left Stick X}};
  \node[smallbox=ctrlcolor, minimum height=14mm, right=6mm of stickcenter, yshift=-12mm]
    (rstick_p) {Pitch\\[-1pt]{\scriptsize\color{gray1} Right Stick Y}};

  \draw[arr=ctrlcolor] (lstick) -- (ctrl);
  \draw[arr=ctrlcolor] (rstick_r) -- (ctrl);
  \coordinate (ctap) at ([yshift=-10mm]ctrl.south);
  \draw[arr=ctrlcolor] (lstick_y.north) |- (ctap) -- (ctrl.south);
  \draw[arr=ctrlcolor] (rstick_p.north) |- (ctap);
  \fill[ctrlcolor] (ctap) circle (2pt);

  % ESP-NOW wireless
  \node[box=espcolor, right=30mm of ctrl] (espnow) {ESP-NOW\\2.4\,GHz};
  \draw[arr=espcolor, decorate, decoration={snake, amplitude=1.5mm, segment length=6mm}]
    (ctrl) -- node[above=3mm, font=\small\bfseries, text=espcolor] {Wireless} (espnow);

  % StampFly
  \node[box=apicolor, right=30mm of espnow] (stampfly) {StampFly};
  \draw[arr=apicolor] (espnow) -- (stampfly);

  % API outputs (2x2 grid with range inside)
  \coordinate (apicenter) at ([yshift=-38mm]stampfly.south);
  \node[smallbox=apicolor, minimum height=14mm, left=6mm of apicenter, yshift=12mm]
    (api1) {rc\_throttle()\\[-1pt]{\scriptsize\color{gray1} 0.0\,--\,1.0}};
  \node[smallbox=apicolor, minimum height=14mm, right=6mm of apicenter, yshift=12mm]
    (api2) {rc\_roll()\\[-1pt]{\scriptsize\color{gray1} -1.0\,--\,+1.0}};
  \node[smallbox=apicolor, minimum height=14mm, left=6mm of apicenter, yshift=-12mm]
    (api3) {rc\_pitch()\\[-1pt]{\scriptsize\color{gray1} -1.0\,--\,+1.0}};
  \node[smallbox=apicolor, minimum height=14mm, right=6mm of apicenter, yshift=-12mm]
    (api4) {rc\_yaw()\\[-1pt]{\scriptsize\color{gray1} -1.0\,--\,+1.0}};

  \draw[arr=apicolor] (stampfly) -- (api1.north);
  \draw[arr=apicolor] (stampfly) -- (api2.north);
  \coordinate (tap) at ([yshift=-10mm]stampfly.south);
  \draw[arr=apicolor] (stampfly.south) -- (tap) -| (api3.north);
  \draw[arr=apicolor] (tap) -| (api4.north);
  \fill[apicolor] (tap) circle (2pt);

\end{tikzpicture}
\end{document}
