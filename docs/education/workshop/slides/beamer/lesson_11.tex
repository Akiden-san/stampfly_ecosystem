\begin{frame}[fragile]{独自ファーム作成}
  \framesubtitle{\texttt{sf app new} でカスタムプロジェクトを生成}
  \begin{block}{\texttt{sf app new} コマンド}
    vehicle のコンポーネントを再利用しつつ、独自の main.cpp を記述
  \end{block}

  \vspace{0.3em}

  \begin{lstlisting}[style=sfbash]
# Create project
sf app new my_drone
# -> firmware/my_drone/ が生成される

# Build and flash
sf build my_drone
sf flash my_drone -m
  \end{lstlisting}

  \vspace{0.3em}
  {\footnotesize
    \textbf{仕組み:} \texttt{EXTRA\_COMPONENT\_DIRS} で vehicle/components と common を参照。\\
    IMU, 気圧, ToF, 光学フロー, モーター等すべてのコンポーネントが使える。
  }
\end{frame}

\begin{frame}{StampFlyState}
  \framesubtitle{全センサ + 推定値への直接アクセス}
  {\small
  \begin{tabular}{lll}
    \hline
    \textbf{メソッド} & \textbf{ソース} & \textbf{取得データ} \\
    \hline
    \texttt{getIMUData(a, g)}       & BMI270 IMU      & 加速度 + ジャイロ生値 \\
    \texttt{getIMUCorrected(a, g)}  & バイアス補正      & バイアス除去済み IMU \\
    \texttt{getBaroData(alt, p)}    & BMP280 気圧      & 高度 [m], 気圧 [Pa] \\
    \texttt{getMagData(mag)}        & BMM150 磁気      & 磁気 x,y,z [$\mu$T] \\
    \texttt{getToFData(b, f)}       & VL53L3CX ToF    & 下方/前方距離 [m] \\
    \texttt{getFlowData(vx, vy)}    & PMW3901 フロー    & 速度 vx,vy [m/s] \\
    \hline
    \texttt{getAttitude()}          & ESKF 推定        & roll, pitch, yaw [rad] \\
    \texttt{getPosition()}          & ESKF 推定        & x, y, z [m] \\
    \texttt{getVelocity()}          & ESKF 推定        & vx, vy, vz [m/s] \\
    \hline
  \end{tabular}
  }
\end{frame}

\begin{frame}[fragile]{実習: StampFlyState で全センサ取得 / Hands-on}
  \framesubtitle{sf app new → StampFlyState で直接アクセス → Teleplot}
  \begin{lstlisting}[style=sfcpp, basicstyle=\ttfamily\scriptsize]
#include "workshop_api.hpp"
#include "stampfly_state.hpp"
auto& state = stampfly::StampFlyState::getInstance();
void setup() { ws::print("L11: StampFlyState"); }
void loop_400Hz(float dt) {
    static uint32_t tick = 0; tick++;
    if (tick % 8 != 0) return;  // 50 Hz
    float alt, p; state.getBaroData(alt, p);
    float bot, fnt; state.getToFData(bot, fnt);
    ws::print(">baro_alt:%.2f", alt);
    ws::print(">baro_pa:%.0f", p);
    ws::print(">tof_bot:%.3f", bot);
}
  \end{lstlisting}

  {\footnotesize
    \textbf{手順:} \texttt{sf app new my\_viz} → main.cpp に記述
    → \texttt{sf build my\_viz} → \texttt{sf flash my\_viz -m}
    → Teleplot で確認
  }
\end{frame}
