\begin{frame}{ジャイロドリフト問題 / Gyro Drift Problem}
  \begin{columns}[c]
    \begin{column}{0.55\textwidth}
      \centering
      \resizebox{!}{0.65\textheight}{\input{../tikz/gyro_drift}}
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{itemize}
        \item ジャイロは\textbf{角速度}を測定
        \item 角度を得るには積分 $\theta = \int \omega\,dt$
        \item バイアス（微小な定常誤差）が蓄積
        \item 時間と共にドリフトが増大
        \item \warn{ジャイロ単体では姿勢推定不可}
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

% ------------------------------------------------------------------
\begin{frame}{相補フィルタ / Complementary Filter}
  \begin{columns}[c]
    \begin{column}{0.55\textwidth}
      \centering
      \resizebox{!}{0.6\textheight}{\input{../tikz/comp_filter}}
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{block}{数式}
        $\hat{\theta}_k = \alpha\,(\hat{\theta}_{k-1} + \omega\,\Delta t) + (1-\alpha)\,\theta_{\mathrm{accel}}$
      \end{block}

      \vspace{0.5em}

      \begin{itemize}
        \item $\alpha = 0.98$（ジャイロ重視）
        \item ジャイロ: 短期精度◎、長期ドリフト✕
        \item 加速度: 長期安定◎、振動ノイズ✕
        \item 両者の長所を融合
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

% ------------------------------------------------------------------
\begin{frame}[fragile]{実習: 相補フィルタ実装 / Hands-on}
  \begin{lstlisting}[style=sfcpp, title={student.cpp (excerpt)}]
#include "workshop_api.hpp"
#include <cmath>
static float cf_roll = 0.0f, cf_pitch = 0.0f;
void setup() { ws::print("Lesson 9: Attitude Estimation"); }
void loop_400Hz(float dt) {
    float gx = ws::gyro_x(), gy = ws::gyro_y();
    float ax = ws::accel_x(), ay = ws::accel_y(),
          az = ws::accel_z();
    float accel_roll  = atan2f(ay, az);
    float accel_pitch = atan2f(-ax, az);
    constexpr float alpha = 0.98f;
    cf_roll  = alpha*(cf_roll  + gx*dt)
             + (1-alpha)*accel_roll;
    cf_pitch = alpha*(cf_pitch + gy*dt)
             + (1-alpha)*accel_pitch;
    ws::telemetry_send("cf_roll",  cf_roll*57.3f);
    ws::telemetry_send("eskf_roll",
                       ws::estimated_roll()*57.3f);
}
  \end{lstlisting}
\end{frame}
