\begin{frame}{ws:: API 一覧 / ws:: API Reference}
  \resizebox{\linewidth}{!}{%
  \begin{tabular}{llp{35mm}}
    \hline
    \textbf{カテゴリ} & \textbf{関数} & \textbf{説明} \\
    \hline
    Motor      & \texttt{motor\_set\_duty(m,d)}, \texttt{motor\_mixer(t,r,p,y)} & モータ制御 \\
    RC Input   & \texttt{rc\_throttle/roll/pitch/yaw()} & スティック値 $[-1,1]$ \\
    Buttons    & \texttt{rc\_throttle\_yaw\_button()}, \texttt{rc\_stabilize\_acro\_mode()} & ボタン/モード \\
    LED        & \texttt{led\_color(r,g,b)}, \texttt{disable\_led\_task()} & LED 制御 \\
    IMU        & \texttt{gyro\_x/y/z()}, \texttt{accel\_x/y/z()} & 角速度, 加速度 \\
    Estimation & \texttt{estimated\_roll/pitch/yaw/altitude()} & ESKF 推定値 \\
    Utility    & \texttt{millis()}, \texttt{battery\_voltage()}, \texttt{print()} & 時刻, 電圧, 出力 \\
    \hline
  \end{tabular}
  }

\begin{frame}[fragile]{独自ファーム作成: \texttt{sf app new}}
  \begin{block}{\api{sf app new} でカスタムプロジェクトを生成}
    vehicle のコンポーネントを再利用しつつ、独自の main.cpp を記述
  \end{block}

  \vspace{0.3em}

  \begin{lstlisting}[style=sfbash]
# Create project / プロジェクト作成
sf app new my_drone
# -> firmware/my_drone/ が生成される

# Build and flash / ビルド & 書き込み
sf build my_drone
sf flash my_drone -m
  \end{lstlisting}

  \vspace{0.3em}
  {\footnotesize
    \textbf{仕組み:} \texttt{EXTRA\_COMPONENT\_DIRS} で vehicle/components と common を参照。\\
    IMU, 気圧, ToF, 光学フロー, モーター等すべてのコンポーネントが使える。
  }
\end{frame}

% ------------------------------------------------------------------
\begin{frame}[fragile]{Teleplot によるセンサ可視化}
  \begin{lstlisting}[style=sfcpp, basicstyle=\ttfamily\scriptsize, title={センサ + ESKF 推定値を Teleplot で可視化}]
auto& s = StampFlyState::getInstance();
auto baro = s.getBaroData();
auto tof  = s.getToFData(ToFPosition::BOTTOM);
auto mag  = s.getMagData();
auto att  = s.getAttitude();   // ESKF roll,pitch,yaw [rad]
float alt = s.getAltitude();   // ESKF estimated altitude [m]

// Teleplot output (>name:value format)
printf(">baro_alt:%.2f\n", baro.altitude);
printf(">tof_bottom:%.3f\n", tof.distance);
printf(">eskf_alt:%.2f\n", alt);
printf(">eskf_roll:%.1f\n", att.x * 57.3f);
printf(">mag_x:%.1f\n", mag.x);
  \end{lstlisting}

  {\footnotesize 気圧高度 vs ToF vs ESKF 推定高度の比較、センサ生値と推定値の違いを観察}
\end{frame}

\begin{frame}[fragile]{実習: 全センサ読み取り / Hands-on}
  {\small ワークショップ版（\texttt{sf lesson switch 10}）のコード例:}

  \begin{lstlisting}[style=sfcpp, basicstyle=\ttfamily\scriptsize, title={user\_code.cpp (excerpt)}]
#include "workshop_api.hpp"
#include "stampfly_state.hpp"
void setup() { ws::print("Lesson 10: API Overview"); }
void loop_400Hz(float dt) {
    static uint32_t tick = 0; tick++;
    if (tick % 8 != 0) return;  // 50 Hz decimation
    // ws:: API for ESKF estimation
    ws::print(">eskf_roll:%.1f", ws::estimated_roll()*57.3f);
    ws::print(">eskf_alt:%.2f", ws::estimated_altitude());
    // StampFlyState for raw sensor data
    auto& s = StampFlyState::getInstance();
    auto baro = s.getBaroData();
    ws::print(">baro_alt:%.2f", baro.altitude);
    auto tof = s.getToFData(ToFPosition::BOTTOM);
    ws::print(">tof_bottom:%.3f", tof.distance);
}
  \end{lstlisting}
